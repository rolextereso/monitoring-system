        $(document).ready(function(){
                  var datapoint = [];
                  var object={
                    type:"line",
                    dataPoints:[]
                  }

                 
                  $.getJSON("samplejson.php", function(data) {
                      object.title=data.title;
                      $.each(data.data, function(key, value){                        

                             object.dataPoints.push({x:  value[0], y: parseInt(value[1])});
                        
                          
                      });

                      datapoint.push(object);
                    
                      var chart = new CanvasJS.Chart("chartContainer",{
                          title:{
                              text:"Rendering Chart with dataPoints from External JSON"
                          },
                          data: datapoint
                      });
                      chart.render();
                  });
              });
